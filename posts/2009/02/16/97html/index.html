def trans(s1, s2=""):
   for i in s1:
       if i in " .,'":
           s2 += i
       elif i is "y":
           s2 += "a"
       elif i is "z":
           s2 += "b"
       else:
           s2 += chr(ord(i) + 2 )
   return s2

print trans(s)
